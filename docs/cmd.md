```bash
# 1. 将文件添加到 .gitignore
echo "要删除的文件名" >> .gitignore

# 2. 从git索引中移除文件
git rm --cached 文件名

# 3. 提交更改
git commit -m "从版本控制移除文件"

# 4. 推送到远程
git push origin dev
```

```bash
# 从暂存区移除特定文件
git reset HEAD 文件名

# 或者移除所有暂存的文件
git reset HEAD

# 然后提交并推送
git commit -m "移除文件"
git push origin 分支名
```

```bash
# 检查文件状态
git status

# 查看远程文件是否已删除
```
